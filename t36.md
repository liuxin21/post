# 函数的属性
  
```js
function a(){
    var va = "xin";
    this.vb = "xin"; 
}
a.vc = "xin";
a.prototype.vd = "xin";


// alert(a.va); // undefined
// alert(a.vb); // undefined
alert(a.vc); //函数的属性
// alert(a.vd); // undefined

var v1 = new a();
// alert(v1.va); // undefined
alert(v1.vb); // 此时a为构造函数
// alert(v1.vc); // undefined
alert(v1.vd); // 此时a为构造函数
```
